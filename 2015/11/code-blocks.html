function codestyling() {
    $('pre code').each(function(i, e) {
        var code = $(this);
        var lines = code.html().split(/\n/).length;
        var numbers = [];
        for (i = 1; i < lines; i++) {
            numbers += '<span class="line">' + i + '</span>';
        }
        code.parent().append('<div class="lines">' + numbers + '</div>');
    });
}
codestyling();
